ll C(ll n, ll r)
{
   if(r<0||r>n)return 0;
   else
   {
       ll val=((((fact[n]%mod)*(ifact[n-r]%mod))%mod)*(ifact[r]%mod))%mod;
       return val;
   }
}
void pre()
{ 
   fact[0]=1;
   ifact[0]=1;//take care
   for(ll i=1;i<=300003;i++)
   {
       fact[i]=fact[i-1]*i;
       fact[i]%=mod;
       ifact[i]=power(fact[i],mod-2,mod);
       ifact[i]%=mod;
   }
}
