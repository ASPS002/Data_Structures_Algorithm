ll power(ll x, ll y, ll p)
{
	ll res = 1;

	x = x % p;

	if (x == 0)
		return 0;
	while (y > 0)
	{

		if (y & 1)
			res = (res * x) % p;

		y = y >> 1;
		x = (x * x) % p;
	}

	return res;
}
void solve()
{
	ll n, k;
	cin >> n;
	k=1;
	ordered_set<ll> ose;
	if (k == 0)
	{
		for (ll i = 0; i < n; i++)
		{
			cout << i + 1 << " ";
		}
		cout << endl;
		return;
	}
	for (ll i = 1; i <= n; i++)
	{
		ose.insert(i);
	}
	ll p = k % n;
	while (ose.size())
	{
		cout << *ose.find_by_order(p) << " ";
		ose.erase(*ose.find_by_order(p));
		if (ose.size())
			p = (p + k) % ose.size();
	}
	cout << endl;
}
